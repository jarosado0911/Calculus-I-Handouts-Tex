\subsection*{Problem 1: Evaluate the Limit if it exists}
\begin{enumerate}[label=(\alph*)]
    \item $\displaystyle\lim_{x\rightarrow-1}\frac{2x^2+3x+1}{x^2-2x-3}$\vspace{3cm}
    \item $\displaystyle\lim_{u\rightarrow 2}\frac{\sqrt{4u+1}-3}{u-2}$\vspace{3cm}
    \item $\displaystyle\lim_{\theta\rightarrow\pi/6}\frac{\sin\theta-(1/2)}{\theta-(\pi/6)}$\vspace{3cm}
\end{enumerate}
